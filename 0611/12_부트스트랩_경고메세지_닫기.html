1. 스크립트로 경고메세지 띄우기

  1. 함수를 만들어서 한 번에 호출

    function msg(){ alert('경고창 띄우기'); }
  2. id 만들어서 호출

    const btn1 = document.getElementById('btn01');
    const btn2 = document.getElementById('btn02');
    btn1.addEventListener('click', msg1);
    btn2.addEventListener('click', msg1);
    function msg1(){
    alert('경고창 띄우기');
    }
  3. class 만들어서 호출

    const c_btn = document.querySelectorAll('.btn03');
    c_btn.forEach(function(button) {
    button.addEventListener('click', function(){
    alert('경고창 띄우기');
    });
    });
